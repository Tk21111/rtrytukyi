cmake_minimum_required(VERSION 3.20)
project(bluetooth_speaker_sync)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Create directories
file(MAKE_DIRECTORY "${CMAKE_SOURCE_DIR}/test/sound")
file(MAKE_DIRECTORY "${CMAKE_SOURCE_DIR}/log")

add_executable(bluetooth_sync
    src/main.cpp
    src/audio_device.cpp
    src/audio_file.cpp
    src/hresult_utils.cpp
    src/logger.cpp
)

# Link Windows multimedia libraries
target_link_libraries(bluetooth_sync
    PRIVATE
    ole32
    oleaut32
    uuid
    winmm
)

# Set output directory
set_target_properties(bluetooth_sync PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# Copy test file if it exists (optional)
if(EXISTS "${CMAKE_SOURCE_DIR}/test/sound/test1.wav")
    add_custom_command(TARGET bluetooth_sync POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_SOURCE_DIR}/test/sound/test1.wav"
        "${CMAKE_BINARY_DIR}/bin/test/sound/test1.wav"
    )
endif()
